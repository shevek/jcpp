017package org.anarres.cpp;
018
019/**
020 * Features of the Preprocessor, which may be enabled or disabled.
021 */
022public enum Feature {
023
024    /** Supports ANSI digraphs. */
025    DIGRAPHS,
026    /** Supports ANSI trigraphs. */
027    TRIGRAPHS,
028    /** Outputs linemarker tokens. */
029    LINEMARKERS,
030    /** Reports tokens of type INVALID as errors. */
031    CSYNTAX,
032    /** Preserves comments in the lexed output. */
033    KEEPCOMMENTS,
034    /** Preserves comments in the lexed output, even when inactive. */
035    KEEPALLCOMMENTS,
036    DEBUG,
037    /** Supports lexing of objective-C. */
038    OBJCSYNTAX,
039    INCLUDENEXT,
040    /** Random extensions. */
041    PRAGMA_ONCE
042}
